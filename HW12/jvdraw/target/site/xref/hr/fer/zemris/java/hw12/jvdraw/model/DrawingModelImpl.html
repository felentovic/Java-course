3   import hr.fer.zemris.java.hw12.jvdraw.geometricalobjects.GeometricalObject;
4   
5   import java.util.ArrayList;
6   import java.util.LinkedList;
7   import java.util.List;
8   
9   /**
10   * Class implements {@link DrawingModel} and has inter list of listeners and
11   * stored objects.
12   * 
13   * @author Borna
14   *
15   */
16  public class DrawingModelImpl implements DrawingModel {
17  	/**
18  	 * Stored objects.
19  	 */
20  	private List<GeometricalObject> objects = new ArrayList<>();
21  	/**
22  	 * List of listeners
23  	 */
24  	private List<DrawingModelListener> listeners = new LinkedList<>();
25  
26  	@Override
27  	public int getSize() {
28  		return objects.size();
29  	}
30  
31  	@Override
32  	public GeometricalObject getObject(int index) {
33  		if (index < 0 || index > objects.size() - 1) {
34  			throw new IllegalArgumentException("Invalid index: " + index);
35  		}
36  		return objects.get(index);
37  	}
38  
39  	@Override
40  	public void add(GeometricalObject object) {
41  		objects.add(object);
42  		for (DrawingModelListener listener : listeners) {
43  			listener.objectsAdded(this, objects.size() - 1, objects.size() - 1);
44  		}
45  	}
46  
47  	@Override
48  	public void addDrawingModelListener(DrawingModelListener l) {
49  		listeners.add(l);
50  	}
51  
52  	@Override
53  	public void removeDrawingModelListener(DrawingModelListener l) {
54  		listeners.remove(l);
55  	}
56  
57  	@Override
58  	public void clear() {
59  		int size = objects.size() == 0 ? 0 : objects.size() - 1;
60  		objects.clear();
61  		for (DrawingModelListener listener : listeners) {
62  			listener.objectsRemoved(this, 0, size);
63  		}
64  	}
65  
66  	@Override
67  	public void addAll(List<GeometricalObject> list) {
68  		int size = objects.size();
69  		objects.addAll(list);
70  		for (DrawingModelListener listener : listeners) {
71  			listener.objectsAdded(this, size, objects.size() - 1);
72  		}
73  	}
74  
75  }
