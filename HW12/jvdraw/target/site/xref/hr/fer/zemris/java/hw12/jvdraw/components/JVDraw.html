300 	/**
301 	 * Asks user to save file. Returns false if action is canceled, otherwise
302 	 * true.
303 	 * 
304 	 * @return false if action is canceled, otherwise true.
305 	 * 
306 	 */
307 	public boolean askToSave() {
308 		int r = JOptionPane.showConfirmDialog(JVDraw.this,
309 				"File is modified. Do you want to save it?", "Warning",
310 				JOptionPane.YES_NO_CANCEL_OPTION, JOptionPane.QUESTION_MESSAGE);
311 		if (r == JOptionPane.CANCEL_OPTION || r == JOptionPane.CLOSED_OPTION) {
312 			return false;
313 
314 		} else if (r == JOptionPane.YES_OPTION) {
315 			save.actionPerformed(null);
316 		}
317 		return true;
318 	}

334 	/**
335 	 * Returns minimal image dimensions. Starting and ending positions.
336 	 * 
337 	 * @return minimal image dimensions
338 	 */
339 	public ImageDimension getImageDimension() {
340 		int length = model.getSize();
341 		if (length == 0) {
342 			return new ImageDimension();
343 		}
344 		ImageDimension dimension = new ImageDimension(model.getObject(0));
345 
346 		for (int i = 1; i < length; i++) {
347 			GeometricalObject object = model.getObject(i);
348 			if (dimension.minX > object.getMinX()) {
349 				dimension.minX = object.getMinX();
350 			}
351 			if (dimension.maxX < object.getMaxX()) {
352 				dimension.maxX = object.getMaxX();
353 			}
354 			if (dimension.minY > object.getMinY()) {
355 				dimension.minY = object.getMinY();
356 			}
357 
358 			if (dimension.maxY < object.getMaxY()) {
359 				dimension.maxY = object.getMaxY();
360 			}
361 		}
362 		return dimension;
363 	}

409 	/**
410 	 * Adds parsed given file content to model. First clears model.
411 	 * 
412 	 * @param file
413 	 *            given file
414 	 * @throws IllegalArgumentException
415 	 *             if content can not be parsed
416 	 */
417 	public void addToModel(String file) {
418 		List<GeometricalObject> objects = new LinkedList<>();
419 		String[] lines = file.split("\n");
420 		int circleCounter = 0;
421 		int filledCcounter = 0;
422 		int lineCounter = 0;
423 		for (String line : lines) {
424 			if (line.startsWith("CIRCLE")) {
425 				circleCounter++;
426 				GeometricalObject object = Circle.parseFromString(line,
427 						"Circle" + circleCounter);
428 				objects.add(object);
429 			} else if (line.startsWith("LINE")) {
430 				lineCounter++;
431 				GeometricalObject object = Line.parseFromString(line, "Line"
432 						+ lineCounter);
433 				objects.add(object);
434 			} else if (line.startsWith("FCIRCLE")) {
435 				filledCcounter++;
436 				GeometricalObject object = FilledCircle.parseFromString(line,
437 						"Filled circle" + filledCcounter);
438 				objects.add(object);
439 			} else {
440 				throw new IllegalArgumentException("Invalid line " + line);
441 			}
442 		}
443 		model.clear();
444 		model.addAll(objects);
445 		JVDraw.this.circleCounter = circleCounter;
446 		JVDraw.this.filledCcounter = filledCcounter;
447 		JVDraw.this.lineCounter = lineCounter;
448 	}

450 	@Override
451 	public GeometricalObject getGeometricalObject(int x, int y) {
452 		String selButt = buttons.getSelection().getActionCommand();
453 		if (selButt.equals("circle")) {
454 			circleCounter++;
455 			return new Circle(x, y, 0, foreground.getCurrentColor(), "Circle"
456 					+ circleCounter);
457 		} else if (selButt.equals("line")) {
458 			lineCounter++;
459 			return new Line(x, y, x, y, foreground.getCurrentColor(), "Line"
460 					+ lineCounter);
461 		} else {
462 			filledCcounter++;
463 			return new FilledCircle(x, y, 0, background.getCurrentColor(),
464 					foreground.getCurrentColor(), "FilledCircle"
465 							+ filledCcounter);
466 		}
467 	}
