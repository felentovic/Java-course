94  		super.paint(g);
95  		Graphics2D g2d = (Graphics2D) g.create();
96  		g2d.setColor(color);
97  		g2d.drawLine(x1 - offsetX, y1 - offsetY, x2 - offsetX, y2 - offsetY);
98  		g2d.dispose();

217 	/**
218 	 * Returns new Line parsed from string.
219 	 * 
220 	 * @param str
221 	 *            string that is parsed
222 	 * @param name
223 	 *            Line name
224 	 * @return new Line parsed from string
225 	 * @throws IllegalArgumentException
226 	 *             if Line cannot be parsed form string
227 	 */
228 	public static GeometricalObject parseFromString(String str, String name) {
229 		String[] elements = str.trim().split("\\s+");
230 		if (elements.length != 8) {
231 			throw new IllegalArgumentException(
232 					"Invalid number of arguments for line: " + str);
233 		}
234 
235 		if (!elements[0].equals("LINE")) {
236 			throw new IllegalArgumentException("Not line: " + str);
237 		}
238 
239 		Line line = null;
240 		try {
241 			int x1 = Integer.parseInt(elements[1]);
242 			int y1 = Integer.parseInt(elements[2]);
243 			int x2 = Integer.parseInt(elements[3]);
244 			int y2 = Integer.parseInt(elements[4]);
245 
246 			int r = Integer.parseInt(elements[5]);
247 			int g = Integer.parseInt(elements[6]);
248 			int b = Integer.parseInt(elements[7]);
249 			Color color = new Color(r, g, b);
250 			line = new Line(x1, y1, x2, y2, color, name);
251 		} catch (NumberFormatException e) {
252 			throw new IllegalArgumentException("Not integer numbers: " + str);
253 		}
254 		return line;
255 	}
