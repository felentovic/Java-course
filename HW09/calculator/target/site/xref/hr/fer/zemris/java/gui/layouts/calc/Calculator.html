44  	 * Constructs new calculator on location (20,50)
45  	 */
46  	public Calculator() {
47  		setLocation(20, 50);
48  		setTitle("Calculator");
49  		setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
50  		initGUI();
51  	}

53  	/**
54  	 * Method used for GUI initialization. Creates buttons and action listeners.
55  	 */
56  	private void initGUI() {
57  
58  		JPanel p = new JPanel(new CalcLayout(3));
59  		JLabel label = new JLabel("0.0");
60  		label.setHorizontalAlignment(SwingConstants.RIGHT);
61  		label.setOpaque(true);
62  		label.setBackground(new Color(249, 206, 36));
63  
64  		CalcMemory memory = new CalcMemory();
65  
66  		ActionListener action = e -> {
67  			IButton button = (IButton) e.getSource();
68  			if (memory.resetNedded
69  					&& !((JButton) button).getText().equals("res")) {
70  				return;
71  			}
72  			button.work(label, memory);
73  		};
74  
75  		FunctionButton[] functions = new FunctionButton[] {
76  				new FunctionButton("1/x", "1/x"),
77  				new FunctionButton("sin", "asin"),
78  				new FunctionButton("log", "10^x"),
79  				new FunctionButton("cos", "acos"),
80  				new FunctionButton("ctg", "actg"),
81  				new FunctionButton("ln", "e^x"),
82  				new FunctionButton("tan", "atan") };
83  
84  		OperatorButton potention = new OperatorButton("x^n");
85  		potention.setInverzName("n\u221Ax");
86  		ActionListener lnvAction = e -> {
87  			JCheckBox checkBox = (JCheckBox) e.getSource();
88  			if (checkBox.isSelected()) {
89  				paintSecondScenario(functions, potention);
90  			} else {
91  				paintFirstScenario(functions, potention);
92  			}
93  
94  		};
95  
96  		p.add(label, "1,1");
97  
98  		p.add(addActionListener(new OperatorButton("="), action), "1,6");
99  		p.add(addActionListener(new ClearButton("clr"), action), "1,7");
100 
101 		p.add(addActionListener(new DigitButton("7"), action), "2,3");
102 		p.add(addActionListener(new DigitButton("8"), action), "2,4");
103 		p.add(addActionListener(new DigitButton("9"), action), "2,5");
104 		p.add(addActionListener(new OperatorButton("/"), action), "2,6");
105 		p.add(addActionListener(new ClearButton("res"), action), "2,7");
106 
107 		p.add(addActionListener(new DigitButton("4"), action), "3,3");
108 		p.add(addActionListener(new DigitButton("5"), action), "3,4");
109 		p.add(addActionListener(new DigitButton("6"), action), "3,5");
110 		p.add(addActionListener(new OperatorButton("*"), action), "3,6");
111 		p.add(addActionListener(new StackButton("push"), action), "3,7");
112 
113 		p.add(addActionListener(new DigitButton("1"), action), "4,3");
114 		p.add(addActionListener(new DigitButton("2"), action), "4,4");
115 		p.add(addActionListener(new DigitButton("3"), action), "4,5");
116 		p.add(addActionListener(new OperatorButton("-"), action), "4,6");
117 		p.add(addActionListener(new StackButton("pop"), action), "4,7");
118 
119 		p.add(addActionListener(new DigitButton("0"), action), "5,3");
120 		p.add(addActionListener(new FunctionButton("+/-", "+/-"), action),
121 				"5,4");
122 		p.add(addActionListener(new DigitButton("."), action), "5,5");
123 		p.add(addActionListener(new OperatorButton("+"), action), "5,6");
124 
125 		p.add(addActionListener(functions[0], action), "2,1");
126 		p.add(addActionListener(functions[1], action), "2,2");
127 		p.add(addActionListener(functions[2], action), "3,1");
128 		p.add(addActionListener(functions[3], action), "3,2");
129 		p.add(addActionListener(potention, action), "5,1");
130 		p.add(addActionListener(functions[4], action), "5,2");
131 		p.add(addActionListener(functions[5], action), "4,1");
132 		p.add(addActionListener(functions[6], action), "4,2");
133 
134 		JCheckBox lnv = new JCheckBox("lnv");
135 		lnv.addActionListener(lnvAction);
136 		p.add(lnv, "5,7");
137 
138 		getContentPane().add(p);
139 		pack();
140 		setMinimumSize(getMinimumSize());
141 
142 	}

144 	/**
145 	 * Method changes button name if {@link Checkbox} is pressed.
146 	 * 
147 	 * @param functions
148 	 *            functions to be changed
149 	 * @param potention
150 	 *            potention to be changed
151 	 */
152 	private void paintSecondScenario(FunctionButton[] functions,
153 			OperatorButton potention) {
154 		for (FunctionButton function : functions) {
155 			function.setText(function.getInverzName());
156 		}
157 		potention.setText(potention.getInverzName());
158 
159 	}
160 
161 	/**
162 	 * Method changes button names on their inverz name.
163 	 * 
164 	 * @param functions
165 	 *            functions to be changed
166 	 * @param potention
167 	 *            potention to be changed
168 	 */
169 	private void paintFirstScenario(FunctionButton[] functions,
170 			OperatorButton potention) {
171 		for (FunctionButton function : functions) {
172 			function.setText(function.getRealName());
173 		}
174 		potention.setText(potention.getRealName());
175 	}
176 
177 	/**
178 	 * Method adds {@link ActionListener} and returns given button.
179 	 * 
180 	 * @param button
181 	 *            button whom {@link ActionListener} is added
182 	 * @param action
183 	 *            {@link ActionListener} that is added to given button
184 	 * @return given button
185 	 */
186 	private JButton addActionListener(JButton button, ActionListener action) {
187 		button.addActionListener(action);
188 		return button;
189 	}
