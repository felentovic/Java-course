3   import java.awt.Dimension;
4   
5   import javax.swing.JButton;
6   
7   /**
8    * Abstract class that extendes {@link JButton} and implements {@link IButton}
9    * has one method for initializing component size.
10   * 
11   * @author Borna Feldšar
12   * @version 1.0
13   */
14  public abstract class AbstractButton extends JButton implements IButton {
15  
16  	private static final long serialVersionUID = 1L;
17  	
18  	/**
19  	 * Constructs {@link AbstractButton} with given name
20  	 * @param name given name
21  	 */
22  	public AbstractButton(String name) {
23  		super(name);
24  	}
25  	
26  	/**
27  	 * Initialize components size to squared.
28  	 */
29  	public void initComponent() {
30  		Dimension buttonDim = this.getPreferredSize();
31  		int square = Math.max(buttonDim.height, buttonDim.width);
32  		buttonDim.height = square;
33  		// because of integer division
34  		buttonDim.width = (int) (1.1 * square);
35  		this.setMinimumSize(buttonDim);
36  	}
37  }
