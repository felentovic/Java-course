6   import java.io.IOException;
7   import java.nio.charset.Charset;
8   import java.util.Arrays;
9   import java.util.Collections;
10  
11  /**
12   * Command charsets takes no arguments and lists names of supported charsets for
13   * your Java platform. A single charset name is written per line.
14   * 
15   * @author Borna Feldšar
16   * @version 1.0
17   *
18   */
19  public class CharsetsCommand extends AbstractCommand {
20  	
21  	/**
22  	 * Constructs charset command with command name "charset" and command description.
23  	 */
24  	public CharsetsCommand() {
25  		super(
26  				"charsets",
27  				Collections
28  				.unmodifiableList(Arrays.asList("Command  takes no arguments "
29  						+ "and lists names of supported charsets for your Java platform.")));
30  
31  	}
32  	
33  	
34  	@Override
35  	public ShellStatus executeCommand(Environment e, String command) {
36  
37  		if (!ArgumentsUtility.isEmpty(e, command)) {
38  			return ShellStatus.CONTINUE;
39  
40  		}
41  
42  		try {
43  			e.writeln("Available charsets for this Java platform are:");
44  		} catch (IOException ignorable) {
45  		}
46  		for (String str : Charset.availableCharsets().keySet()) {
47  			try {
48  				e.writeln(str);
49  			} catch (IOException ignorable) {
50  			}
51  		}
52  
53  		return ShellStatus.CONTINUE;
54  	}
55  
56  }
