7   /**
8    * Class extendes {@link AbstractMatrix} and implements unimplemented methods.
9    * 
10   * @author Borna Feldšar
11   * @version 1.0
12   */
13  public class Matrix extends AbstractMatrix {
14  	private double[][] elements;
15  	private int rows;
16  	private int cols;
17  
18  	/**
19  	 * Constructs empty matrix with given number of rows and columns
20  	 * 
21  	 * @param rows
22  	 *            number of rows
23  	 * @param columns
24  	 *            number of columns
25  	 * @throws IllegalArgumentException
26  	 *             if rows or columns are smaller than one
27  	 */
28  	public Matrix(int rows, int columns) {
29  		this(rows, columns, getArrayRowsCols(rows, columns), true);
30  
31  	}
32  
33  	/**
34  	 * Constructs matrix from given array. If flag use given is true this
35  	 * elements and elements share same reference.
36  	 * 
37  	 * @param rows
38  	 *            number of used rows
39  	 * @param cols
40  	 *            number of used columns
41  	 * @param elements2
42  	 *            elements array
43  	 * @param useGiven
44  	 *            if true this elemenents and elements share same reference,
45  	 *            otherwise array is copied
46  	 */
47  	public Matrix(int rows, int cols, double[][] elements2, boolean useGiven) {
48  		checkRowsCols(rows, cols);
49  		this.rows = rows;
50  		this.cols = cols;
51  		checkArray(rows, cols, elements2);
52  		if (useGiven) {
53  			this.elements = elements2;
54  		} else {
55  			this.elements = new double[rows][cols];
56  			for (int i = 0; i < rows; i++) {
57  				elements[i] = Arrays.copyOf(elements2[i], cols);
58  			}
59  
60  		}
61  	}
62  
63  	/**
64  	 * Trows {@link IllegalArgumentException} if rows length or columns length
65  	 * is bigger than columns and rows lengtth in elements2 array.
66  	 * 
67  	 * @param rows2
68  	 *            used rows length
69  	 * @param cols2
70  	 *            used cols length
71  	 * @param elements2
72  	 *            elements array
73  	 */
74  	private void checkArray(int rows2, int cols2, double[][] elements2) {
75  		if (rows2 > elements2.length) {
76  			throw new IllegalArgumentException("Given array of elements has"
77  					+ " smaller number of elements than given in constructor.");
78  		}
79  		for (int i = 0; i < rows2; i++) {
80  			if (cols2 > elements2[i].length) {
81  				throw new IllegalArgumentException(
82  						"Given array of elements has"
83  								+ " smaller number of elements than given in constructor.");
84  			}
85  		}
86  
87  	}
88  
89  	@Override
90  	public IMatrix copy() {
91  		IMatrix matrix = newInstance(rows, cols);
92  		for (int i = 0; i < rows; i++) {
93  			for (int j = 0; j < cols; j++) {
94  				matrix.set(i, j, elements[i][j]);
95  			}
96  		}
97  		return matrix;
98  	}

101 	 * @throws IllegalArgumentException

103 	 */

105 	public double get(int row, int col) {
106 		checkIndex(row, col);

121 	public IMatrix newInstance(int rows, int cols) {

126 	 * @throws IllegalArgumentException

129 	@Override
130 	public IMatrix set(int row, int col, double value) {

136 	/**
137 	 * Parse string into new matrix and returns it. i.e. 1 2 3 | 3 4 5
138 	 * 
139 	 * @param str
140 	 *            string that is parsed
141 	 * @return new matrix from string
142 	 * @throws IllegalArgumentException
143 	 *             if element can not be parsed into double
144 	 * @throws IllegalArgumentException
145 	 *             if columns are not same length
146 	 */
147 	public static Matrix parseString(String str) {
148 		String[] rows = str.split("\\|");
149 		LinkedList<List<Double>> elem = new LinkedList<>();
150 
151 		for (int row = 0; row < rows.length; row++) {
152 			String[] cols = rows[row].trim().split(" ");
153 			List<Double> list = new LinkedList<Double>();
154 
155 			for (int col = 0; col < cols.length; col++) {
156 				try {
157 					list.add(Double.parseDouble(cols[col]));
158 				} catch (NumberFormatException e) {
159 					throw new IllegalArgumentException(
160 							"Can't parse into double.");
161 				}
162 			}
163 
164 			if (row != 0) {
165 				if (elem.getLast().size() != list.size()) {
166 					throw new IllegalArgumentException(
167 							"Columns are not same length");
168 				}
169 			}
170 			elem.add(list);
171 		}
172 		double[][] array = new double[elem.size()][elem.getFirst().size()];
173 		int i = 0, j = 0;
174 		for (List<Double> outer : elem) {
175 			j = 0;
176 			for (double number : outer) {
177 				array[i][j] = number;
178 				j++;
179 			}
180 			i++;
181 		}
182 
183 		return new Matrix(i, j, array, true);
184 	}

186 	/**
187 	 * Throws {@link IllegalArgumentException} if given row or colums is smaller
188 	 * than 0 or bigger than length-1
189 	 * 
190 	 * @param row
191 	 *            given row
192 	 * @param col
193 	 *            given column
194 	 */
195 	private void checkIndex(int row, int col) {
196 		if (row < 0 || row >= rows || col < 0 || col >= cols) {
197 			throw new IllegalArgumentException("Given row:" + row + " or col:"
198 					+ col + " are smaller"
199 					+ "than zero or equal, bigger than matrix dimension.");
200 		}
201 	}
202 
203 	/**
204 	 * Method returns double[][] array from given rows and columns length.
205 	 * 
206 	 * @param rows
207 	 *            rows length
208 	 * @param columns
209 	 *            columns length
210 	 * @return double array
211 	 * @throws IllegalArgumentException
212 	 *             if rows or columns are smaller than 1
213 	 */
214 	private static double[][] getArrayRowsCols(int rows, int columns) {
215 		checkRowsCols(rows, columns);
216 		return new double[rows][columns];
217 	}
218 
219 	/**
220 	 * Throws {@link IllegalArgumentException} if rows or columns are smaller
221 	 * than one
222 	 * 
223 	 * @param rows
224 	 *            rows length
225 	 * @param columns
226 	 *            columns length
227 	 */
228 	private static void checkRowsCols(int rows, int columns) {
229 		if (rows < 1 || columns < 1) {
230 			throw new IllegalArgumentException(
231 					"Rows and columns cant be smaller than 1.");
232 		}
233 
234 	}
