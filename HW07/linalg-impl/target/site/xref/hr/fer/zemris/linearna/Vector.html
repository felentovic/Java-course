3   /**
4    * Vector implementation extends {@link AbstractVector}. Provides basic methods
5    * for vector algebra
6    * 
7    * @author Borna Feld≈°ar
8    * @version 1.0
9    */
10  public class Vector extends AbstractVector {
11  
12  	private double[] elements;
13  	private int dimension;
14  	private boolean readOnly;
15  
16  	/**
17  	 * Constructs new vector from given elements array. Vector is not read only.
18  	 * 
19  	 * @param elems
20  	 *            elements array
21  	 */
22  	public Vector(double... elems) {
23  		this(false, false, elems);
24  	}
25  
26  	/**
27  	 * Constucts new vector from given elements array.
28  	 * 
29  	 * @param readOnly
30  	 *            if true this readOnly flag is true, false otherwise
31  	 * @param useGiven
32  	 *            if true vector use same reference on elements array, otherwise
33  	 *            creates new one and copies elements
34  	 * @param elems
35  	 *            elements array
36  	 */
37  	public Vector(boolean readOnly, boolean useGiven, double... elems) {
38  		if (elems == null) {
39  			throw new IllegalArgumentException("Null elements");
40  		}
41  
42  		int length = elems.length;
43  		if (useGiven) {
44  			elements = elems;
45  		} else {
46  			elements = new double[length];
47  			System.arraycopy(elems, 0, elements, 0, length);
48  
49  		}
50  		this.readOnly = readOnly;
51  		dimension = length;
52  	}
53  
54  	/**
55  	 * Constructs vector with given dimension.
56  	 * 
57  	 * @param dimension
58  	 *            initial dimension
59  	 */
60  	public Vector(int dimension) {
61  		if (dimension < 1) {
62  			throw new IllegalArgumentException("Given dimension :" + dimension
63  					+ " is smaller than one.");
64  		}
65  		elements = new double[dimension];
66  		readOnly = false;
67  		this.dimension = dimension;
68  	}
69  
70  	/**
71  	 * Return this vector with same atributtes but different reference.
72  	 */
73  	@Override
74  	public IVector copy() {
75  		return copyPart(dimension);
76  	}
77  
78  	/**
79  	 * Returns element on given index.
80  	 * 
81  	 * @throws IllegalArgumentException
82  	 *             if index is smaller than zero or bigger than vetor
83  	 *             dimension-1
84  	 */
85  	@Override
86  	public double get(int index) {
87  		checkIndex(index);
88  		return elements[index];
89  	}
90  
91  	/**
92  	 * Returns vector dimension.
93  	 */
94  	@Override
95  	public int getDimension() {
96  		return dimension;
97  	}
98  
99  	/**
100 	 * Returns new instance of vector with given dimension.
101 	 * 
102 	 * @throws IllegalArgumentException
103 	 *             if dimesion is smaller than 1
104 	 */
105 	@Override
106 	public IVector newInstance(int dimension) {
107 		return new Vector(dimension);
108 
109 	}
110 
111 	/**
112 	 * Sets new element value on given index.
113 	 * 
114 	 * @throws IllegalArgumentException
115 	 *             if index is smaller than zero or bigger than vetor
116 	 *             dimension-1
117 	 */
118 	@Override
119 	public IVector set(int index, double value)
120 			throws UnmodifiableObjectException {
121 		if (readOnly) {
122 			throw new UnmodifiableObjectException("Vector is read only.");
123 		}
124 		checkIndex(index);
125 		elements[index] = value;
126 		return this;
127 	}
128 
129 	/**
130 	 * Return new vector parsed from string. In string are only vetor elements.
131 	 * 
132 	 * @param str
133 	 *            vector elements
134 	 * @return new vector
135 	 * @throws IllegalArgumentException
136 	 *             if elements can't be parsed into double
137 	 * @throws IllegalArgumentException
138 	 *             if string is empty
139 	 */
140 	public static Vector parseSimple(String str) {
141 		if (str == null) {
142 			throw new IllegalArgumentException("String is null.");
143 		}
144 
145 		String[] elems = str.trim().split("\\s+");
146 		if (elems[0].isEmpty()) {
147 			throw new IllegalArgumentException("Given empty string.");
148 		}
149 		double[] elem = new double[elems.length];
150 
151 		for (int i = 0; i < elems.length; i++) {
152 			try {
153 				elem[i] = Double.parseDouble(elems[i]);
154 			} catch (NumberFormatException e) {
155 				throw new IllegalArgumentException(
156 						"Given element cant be parsed into double " + elems[i]);
157 			}
158 		}
159 		return new Vector(elem);
160 	}
161 
162 	private void checkIndex(int index) {
163 		if (index < 0 || index >= dimension) {
164 			throw new IllegalArgumentException("Invalid index :" + index);
165 		}
166 	}
167 }
