11  public abstract class AbstractVector implements IVector {
12  
13  	/**
14  	 * Adds given vector to this vector, and returns this vector.
15  	 * 
16  	 * @param other
17  	 *            vector that is added to this vector
18  	 * @return this vector
19  	 * @throws IncompatibleOperandException
20  	 *             if vectors are not same dimension.
21  	 */
22  	@Override
23  	public IVector add(IVector other) throws IncompatibleOperandException {
24  		sameDimensions(other);
25  		for (int i = 0; i < this.getDimension(); i++) {
26  			this.set(i, this.get(i) + other.get(i));
27  		}
28  		return this;
29  	}
30  
31  	/**
32  	 * Creates new vector with given dimension. If dimension is smaller than
33  	 * this dimension elements are copied till dimension, otherwise elements are
34  	 * zero.
35  	 * 
36  	 * @param dimension
37  	 *            dimension of new vector
38  	 * @return new vector with given dimension and copied elements
39  	 * @throws IllegalArgumentException
40  	 *             if dimension is smaller than zero
41  	 */
42  	@Override
43  	public IVector copyPart(int dimension) {
44  		IVector vector = newInstance(dimension);
45  		for (int i = 0; i < dimension; i++) {
46  			if (i < this.getDimension()) {
47  				vector.set(i, this.get(i));
48  			} else {
49  				vector.set(i, 0);
50  			}
51  		}
52  		return vector;
53  	}
54  
55  	/**
56  	 * Returns cosine from given vector and this vector.
57  	 * 
58  	 * @param other
59  	 *            other vector
60  	 * @return cosine from given vector and this vector.
61  	 * @throws IncompatibleOperandException
62  	 *             if vectors are not same dimension
63  	 */
64  	@Override
65  	public double cosine(IVector other) throws IncompatibleOperandException {
66  		sameDimensions(other);
67  
68  		double thisMagnitude = this.norm();
69  		double otherMagnitude = other.norm();
70  		if (thisMagnitude == 0 || otherMagnitude == 0) {
71  			throw new IncompatibleOperandException(
72  					"This or other vector norm is zero");
73  		}
74  		return this.scalarProduct(other) / (thisMagnitude * otherMagnitude);
75  	}

93  	/**
94  	 * Returns new homogenus vector with new dimension= this dimension -1. All
95  	 * elements are divided with the last element from this vector.
96  	 * 
97  	 * @return new vector
98  	 */
99  	@Override
100 	public IVector nFromHomogeneus() {
101 		if (this.getDimension() == 1) {
102 			throw new IncompatibleOperandException("Elements dimension is one.");
103 		}
104 		if (this.get(getDimension() - 1) == 0) {
105 			throw new IncompatibleOperandException(
106 					"Last element of vector is zero");
107 		}
108 		IVector vector = newInstance(this.getDimension() - 1);
109 		double lastElement = this.get(this.getDimension() - 1);
110 
111 		for (int i = 0; i < vector.getDimension(); i++) {
112 			vector.set(i, this.get(i) / lastElement);
113 		}
114 
115 		return vector;
116 	}

159 	/**
160 	 * Returns new vector which is result of vector product on this vector and
161 	 * given vector.
162 	 * 
163 	 * @param other
164 	 *            other vector
165 	 * @return new vector
166 	 * @throws IncompatibleOperandException
167 	 *             if this or other vector are not three-dimensional
168 	 */
169 	@Override
170 	public IVector nVectorProduct(IVector other)
171 			throws IncompatibleOperandException {
172 		sameDimensions(other);
173 
174 		if (this.getDimension() != 3) {
175 			throw new IncompatibleOperandException(
176 					"Vectors are not three-dimensional.");
177 		}
178 
179 		IVector vector = newInstance(3);
180 		double this1 = this.get(0);
181 		double this2 = this.get(1);
182 		double this3 = this.get(2);
183 
184 		double other1 = other.get(0);
185 		double other2 = other.get(1);
186 		double other3 = other.get(2);
187 		vector.set(0, this2 * other3 - this3 * other2);
188 		vector.set(1, -this1 * other3 + this3 * other1);
189 		vector.set(2, this1 * other2 - this2 * other1);
190 
191 		return vector;
192 	}

194 	/**
195 	 * Returns vectors norm.
196 	 * 
197 	 * @return vectors norm
198 	 */
199 	@Override
200 	public double norm() {
201 		double norm = 0;
202 		for (int i = 0; i < this.getDimension(); i++) {
203 			norm += Math.pow(this.get(i), 2);
204 		}
205 		return Math.sqrt(norm);
206 	}
207 
208 	/**
209 	 * Normalizes this vector to unit vector.
210 	 * 
211 	 * @return this vector
212 	 * @throws UnsupportedOperationException
213 	 *             if vectors norm is zero
214 	 */
215 	@Override
216 	public IVector normalize() {
217 		double norm = this.norm();
218 		if (norm == 0) {
219 			throw new IncompatibleOperandException(
220 					"You cant normalize vector whose norm is zero.");
221 		}
222 		for (int i = 0; i < this.getDimension(); i++) {
223 			this.set(i, this.get(i) / norm);
224 		}
225 
226 		return this;
227 	}

230 	 * Multiplies each element of this vector with given scalar and returns this
231 	 * vector.
232 	 * 
233 	 * @param byValue
234 	 *            scalar
235 	 * @return this vector

237 	@Override
238 	public IVector scalarMultiply(double byValue) {
239 		for (int i = 0; i < this.getDimension(); i++) {
240 			this.set(i, this.get(i) * byValue);
241 		}
242 		return this;
243 	}

245 	/**
246 	 * Multiplies elements of this vector with given vector. First this element
247 	 * * first other element..
248 	 * 
249 	 * @param other
250 	 *            other vector
251 	 * @return scalar product of vectors
252 	 * @throws IncompatibleOperandException
253 	 *             if vectors are not same dimension
254 	 */
255 	@Override
256 	public double scalarProduct(IVector other)
257 			throws IncompatibleOperandException {
258 		sameDimensions(other);
259 
260 		double scalar = 0;
261 		for (int i = 0; i < this.getDimension(); i++) {
262 			scalar += this.get(i) * other.get(i);
263 		}
264 
265 		return scalar;
266 	}

269 	 * Subs this vector and other vector. Returns this vector.
270 	 * 
271 	 * @param other
272 	 *            other vector
273 	 * @return this vector
274 	 * @throws IncompatibleOperandException
275 	 *             if vectors are not same dimension
276 	 */
277 	@Override
278 	public IVector sub(IVector other) throws IncompatibleOperandException {
279 		sameDimensions(other);
280 
281 		for (int i = 0; i < this.getDimension(); i++) {
282 			this.set(i, this.get(i) - other.get(i));
283 		}
284 		return this;
285 	}

330 	/**
331 	 * Returns string representation of vector.
332 	 * 
333 	 * @param n
334 	 *            number of element decimals
335 	 * @return string representation of vector
336 	 */
337 	public String toString(int n) {
338 		if (n < 0) {
339 			throw new IllegalArgumentException(
340 					"Decimals number is smaller than zero :" + n);
341 		}
342 		StringBuilder str = new StringBuilder();
343 		str.append("(");
344 		for (int i = 0; i < this.getDimension(); i++) {
345 			str.append(String.format("%." + n + "f,", this.get(i)));
346 		}
347 		str.deleteCharAt(str.length() - 1);
348 		str.append(")");
349 		return str.toString();
350 	}

352 	/**
353 	 * Throws {@link IncompatibleOperandException} if this vector and other
354 	 * vector are not same dimension.
355 	 * 
356 	 * @param other
357 	 *            other vector
358 	 */
359 	private void sameDimensions(IVector other) {
360 		if (other == null) {
361 			throw new IllegalArgumentException("Other vector is null");
362 		}
363 		if (this.getDimension() != other.getDimension()) {
364 			throw new IncompatibleOperandException(
365 					"Vectors are not same dimension.");
366 		}

369 	@Override
370 	public boolean equals(Object obj) {
371 		if (this == obj)
372 			return true;
373 		if (obj == null)
374 			return false;
375 		if (!(obj instanceof IVector))
376 			return false;
377 		IVector other = (IVector) obj;
378 		if (getDimension() != other.getDimension()) {
379 			return false;
380 		}
381 		int dimension = getDimension();
382 
383 		for (int i = 0; i < dimension; i++) {
384 			if (Math.abs(this.get(i) - other.get(i)) > 0.00000001) {
385 				return false;
386 			}
387 		}
388 
389 		return true;
390 	}
