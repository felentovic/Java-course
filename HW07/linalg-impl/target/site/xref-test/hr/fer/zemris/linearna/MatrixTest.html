8   	@Test
9   	public void parseString_True_EqualMatrix() {
10  		String str = "1 2 3| 4 5 6 |";
11  		IMatrix matrix = Matrix.parseString(str);
12  		double[][] elements = { { 1, 2, 3 }, { 4, 5, 6 } };
13  
14  		Assert.assertEquals(new Matrix(2, 3, elements, true), matrix);
15  	}
16  
17  	@Test
18  	public void toString_True_Output() {
19  		String str = "1 2 3| 4 5 6 |";
20  		IMatrix matrix = Matrix.parseString(str);
21  		String output = "[1,000, 2,000, 3,000]\n" + "[4,000, 5,000, 6,000]\n";
22  		Assert.assertEquals(output, matrix.toString());
23  	}
